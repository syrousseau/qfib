73  template<typename T>
74  Vector3f SF(const Bitvalue<T> id, const uint32_t nbPoints)
75  {
76  double m = 1.0 - 1.0 / nbPoints;
77  double phi = TWO_PI * myfract(id.getCombinedValue() * PHI);
78  double cosTheta = m - 2.0 * double(id.getCombinedValue()) / nbPoints;
79  double sinTheta = sqrt(1.0 - cosTheta * cosTheta);
80  return Vector3f(cos(phi) * sinTheta, sin(phi) * sinTheta, cosTheta).normalized();
81  }

97  template<typename T>
98  T inverseSF(const Vector3f & vec, const unsigned n)
99  {
100  Vector3d p(vec.x(), vec.y(), vec.z());
101  double m = 1.0 - 1.0 / n;
102  double phi = std::min(std::atan2(p.y(), p.x()), PI), cosTheta = p.z();
103  double k = std::max(2.0, floor(log(n * PI * sqrt(5.0) * (1.0 - cosTheta*cosTheta)) / log(PHI * PHI)));
104  double Fk = pow(PHI, k) / sqrt(5.0);
105  Vector2d F = Vector2d(round(Fk), round(Fk * PHI));
106 
107  Vector2d ka = 2.0 * F / double(n);
108  Vector2d kb = TWO_PI *(myfract((F + Vector2d(1.0, 1.0)) * PHI) - Vector2d(PHI - 1.0, PHI - 1.0));
109  Matrix2d iB;
110  iB << ka.y(), kb.y(), -ka.x(), -kb.x();
111  iB /= (ka.y() * kb.x() - ka.x() * kb.y());
112 
113  Vector2d c = myfloor(iB * Vector2d(phi, cosTheta - m));
114  double d = 8.0;
115  T j = 0;
116  for (unsigned s = 0; s < 4; ++s)
117  {
118  Vector2d uv = Vector2d(double(s - 2 * (s / 2)), double(s / 2));
119  double i = F.dot(uv + c);
120 
121  double phi = TWO_PI * myfract(i * PHI);
122  double cosTheta = m - 2.0 * i / n;
123  double sinTheta = sqrt(1.0 - cosTheta*cosTheta);
124 
125  Vector3d q = Vector3d(cos(phi) * sinTheta, sin(phi) * sinTheta, cosTheta);
126  double squaredDistance = (q - p).dot(q - p);
127  if (squaredDistance < d)
128  {
129  d = squaredDistance;
130  j = T(i);
131  }
132  }
133  return j;
134  }
