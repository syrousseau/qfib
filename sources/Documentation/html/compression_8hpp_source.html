97 Vector3f uniformMapping(const Vector3f & v, const Vector3f & axis, const float ratio)
98 {
99  Vector3d dv = v.cast<double>();
100  Vector3d daxis = axis.cast<double>();
101  dv.normalize();
102  daxis.normalize();
103 
104  double K = 1.0 / ratio;
105  double c = dv.dot(daxis);
106 
107  // numerical instabilities
108  if (c > 1.0)
109  c = 1.0;
110  if (c < -1.0)
111  c = -1.0;
112  Vector3d p1 = (dv - c * daxis).normalized();
113  double delta = (1.0 - ((1.0 - c) / K));
114  return ((p1 * std::sqrt(1.0 - delta * delta) + delta * daxis).cast<float>()).normalized();
115 }

125 Vector3f inverseUniformMapping(const Vector3f & v, const Vector3f & axis, const float ratio)
126 {
127  //return v;
128  Vector3d v2 = v.cast<double>();
129  Vector3d axis2 = axis.cast<double>();
130  v2.normalize();
131  axis2.normalize();
132 
133  double K = ratio;
134  double c = v2.dot(axis2);
135 
137  if (c > 1.0)
138  c = 1.0;
139  if (c < -1.0)
140  c = -1.0;
141  Vector3d p1 = (v2 - c * axis2).normalized();
142  double delta = (1.0 - ((1.0 - c) / K));
143  return ((p1 * std::sqrt(1.0 - delta * delta) + delta * axis2).cast<float>()).normalized();
144 }

152 float computeRatio(const std::vector<Vector3f> & fibers)
153 {
154  Vector3f v1, v2;
155  float mindot = 2.f;
156  v1 = (fibers[1] - fibers[0]).normalized();
157  for(unsigned p = 2; p < fibers.size(); ++p)
158  {
159  v2 = (fibers[p] - fibers[p-1]).normalized();
160  float dot = v2.dot(v1);
161  mindot = (dot < mindot) ? dot : mindot;
162  v1 = v2;
163  }
164 
165  return mindot;
166 }

176 float computeRatioVerif(const std::vector<std::vector<Vector3f> > & fibers, bool verbose, uint64_t & totalNbPts)
177 {
178  unsigned maxThreads = std::max(unsigned(1), std::thread::hardware_concurrency());
179  std::vector<uint64_t> nbPts(maxThreads, 0);
180  std::vector<float> mindot(maxThreads, 2.0f);
181  std::vector<float> maxStep(maxThreads, 0.0f);
182  std::vector<float> minStep(maxThreads, std::numeric_limits<float>::max());
183 #ifdef USE_OPENMP
184 #pragma omp parallel for
185  for(int f = 0; f < fibers.size(); ++f)
186  {
187  nbPts[omp_get_thread_num()] += fibers[f].size();
188  Vector3f v1, v2;
189  float v2norm;
190  v1 = (fibers[f][1] - fibers[f][0]).normalized();
191  for(unsigned p = 2; p < fibers[f].size(); ++p)
192  {
193  v2 = (fibers[f][p] - fibers[f][p-1]);
194  v2norm = v2.norm();
195  maxStep[omp_get_thread_num()] = (v2norm > maxStep[omp_get_thread_num()]) ? v2norm : maxStep[omp_get_thread_num()];
196  minStep[omp_get_thread_num()] = (v2norm < minStep[omp_get_thread_num()]) ? v2norm : minStep[omp_get_thread_num()];
197  float dot = (v2 / v2norm).dot(v1);
198  mindot[omp_get_thread_num()] = (dot < mindot[omp_get_thread_num()]) ? dot : mindot[omp_get_thread_num()];
199  v1 = v2 / v2norm;
200  }
201  }
202 #else
203  parallel.For<unsigned>(0, fibers.size(), 1, [&](unsigned f)
204  {
205  nbPts[parallel.getID()] += fibers[f].size();
206  Vector3f v1, v2;
207  float v2norm;
208  v1 = (fibers[f][1] - fibers[f][0]).normalized();
209  for(unsigned p = 2; p < fibers[f].size(); ++p)
210  {
211  v2 = (fibers[f][p] - fibers[f][p-1]);
212  v2norm = v2.norm();
213  maxStep[parallel.getID()] = (v2norm > maxStep[parallel.getID()]) ? v2norm : maxStep[parallel.getID()];
214  minStep[parallel.getID()] = (v2norm < minStep[parallel.getID()]) ? v2norm : minStep[parallel.getID()];
215  float dot = (v2 / v2norm).dot(v1);
216  mindot[parallel.getID()] = (dot < mindot[parallel.getID()]) ? dot : mindot[parallel.getID()];
217  v1 = v2 / v2norm;
218  }
219  });
220 #endif
221  float mdot = mindot[0], maxS = maxStep[0], minS = minStep[0];
222  totalNbPts=nbPts[0];
223  for (unsigned int i=1; i<mindot.size(); i++)
224  {
225  totalNbPts += nbPts[i];
226  mdot = (mindot[i] < mdot) ? mindot[i] : mdot;
227  maxS = (maxStep[i] > maxS) ? maxStep[i] : maxS;
228  minS = (minStep[i] < minS) ? minStep[i] : minS;
229  }
230  float diff = maxS - minS;
231 
232  if (verbose)
233  {
234  std::cout << "Max angle: " << acos(mdot) * 180 / fib::PI << "°" << std::endl;
235  std::cout << "Max length between segments: " << maxS << std::endl;
236  std::cout << "Min length between segments: " << minS << std::endl;
237  std::cout << "Difference: " << diff << std::endl;
238  }
239 
240  if (diff > 0.1f * (maxS + minS) / 2.0f)
241  {
242  std::cerr << std::endl << "-------------!ERROR!--------------" << std::endl;
243  std::cerr << "Difference in segments length too big (" << diff << "), no compression possible" << std::endl;
244  std::cerr << "Try to resample your data with a constant stepsize before compression" << std::endl;
245  std::cerr << "-------------!ERROR!--------------" << std::endl << std::endl;
246  exit(EXIT_FAILURE);
247  }
248 
249  epsMapping = 0.f;
250  for(unsigned i = 0; i < 20; ++i)
251  epsMapping = 3 * sqrt(2) * pow(2, -bitcount/2.f) * ((1 - (mdot))/2+epsMapping);
252  //Error maximization proportionaly to itself (epsMapping < 1)
253  epsMapping = sqrt(epsMapping);
254  return std::min(1.f, (1.f - mdot) / 2.f + epsMapping);
255 }

330 template<typename T>
331 void compressFiber(const std::vector<Vector3f> & fiber, const float ratio, CompressedFiber<T> & cfiber)
332 {
333  assert(fiber.size() > 1);
334 
335  //encode the first two points
336  cfiber.origin = fiber[0];
337  cfiber.second = fiber[1];
338 
339  cfiber.data.resize(fiber.size() - 2);
340 
341  //compute the first unit vector and the stepsize
342  Vector3f axis = (fiber[1] - fiber[0]).normalized();
343  float stepsize = (fiber[1] - fiber[0]).norm();
344  Vector3f currentpoint = fiber[1];
345 
346  //compress the rest of the fiber
347  Vector3f v;
348  constexpr float eps = 0.00000001;
349  for(unsigned i = 2; i < fiber.size(); ++i)
350  {
351  v = (fiber[i] - currentpoint).normalized();
352  if(((1.f - v.dot(axis)) / 2.f) > ratio) //Conservative mapping in case the epsilon is not enough
353  {
354  fc::quantize((-((1.0 - eps) * axis) + (eps * v)).normalized(), cfiber.data[i-2]);
355  }
356  else //Otherwise, general case
357  {
358  if(abs(v.dot(axis)) < 1.f)
359  fc::quantize(fc::inverseUniformMapping(v, axis, ratio), cfiber.data[i-2]);
360  else
361  {
362  fc::quantize(v, cfiber.data[i-2]);
363  }
364  }
365  axis = fc::uniformMapping(fc::unquantize(cfiber.data[i-2]), axis, ratio);
366  currentpoint += (stepsize * axis);
367  //currentpoint = fiber[i]; //Uncomment if you want to disable error propagation !!
368  }
369 }

440  {

444 
445  fiber.push_back(fiber[i+1] + stepsize * v);
446  axis = v;
447  }
