52 float computeTrilinearInterpolatedValue(std::vector<float> const & grid, std::vector<unsigned> const & dim, Vector3f const & gridPosition)
53 {
54  std::vector<float> value;
55  for (unsigned pos2=0; pos2<2; pos2++)
56  for (unsigned pos3=0; pos3<2; pos3++)
57  for (unsigned pos1=0; pos1<2; pos1++)
58  value.push_back(grid[(int)my_round(pos1,gridPosition(0))+dim[0]*(int)my_round(pos2,gridPosition(1))+dim[0]*dim[1]*(int)my_round(pos3,gridPosition(2))]);
59  float xd = gridPosition(0) - floor(gridPosition(0));
60  float yd = gridPosition(1) - floor(gridPosition(1));
61  float zd = gridPosition(2) - floor(gridPosition(2));
62 
63  float c00 = value[0]*(1-xd)+value[1]*xd;
64  float c01 = value[2]*(1-xd)+value[3]*xd;
65  float c10 = value[4]*(1-xd)+value[5]*xd;
66  float c11 = value[6]*(1-xd)+value[7]*xd;
67 
68  float c0 = c00*(1-yd)+c10*yd;
69  float c1 = c01*(1-yd)+c11*yd;
70 
71  float val = c0*(1-zd)+c1*zd;
72 
73  return val;
74 }

104 float getValueFromSegment(std::vector<float> const & grid, std::vector<unsigned> const & dim, Vector3f const & p1, Vector3f const & p2)
105 {
106  Vector3f direction = p2-p1;
107  Vector3f directionN = direction.normalized();
108  float totalDist = direction.norm();
109  //Number of voxels changes along each axis
110  unsigned voxelsX = static_cast<unsigned>(fabs(floor(p2.x())-floor(p1.x())));
111  unsigned voxelsY = static_cast<unsigned>(fabs(floor(p2.y())-floor(p1.y())));
112  unsigned voxelsZ = static_cast<unsigned>(fabs(floor(p2.z())-floor(p1.z())));
113  unsigned totalVoxelsCrossed = 1 + voxelsX + voxelsY + voxelsZ;
114  Vector3f pTemp = p1;
115  float value = 0;
116  float epsilon = 0.0001f;
117  float gammaX, gammaY, gammaZ, gamma;
118 
119  for (unsigned i=0; i<totalVoxelsCrossed - 1; i++)
120  {
121  if (fabs(directionN.x())<epsilon)
122  gammaX = numeric_limits<float>::max();
123  else
124  gammaX = (k2(pTemp.x(), direction.x()) - pTemp.x()) / direction.x();
125  if (fabs(directionN.y())<epsilon)
126  gammaY = numeric_limits<float>::max();
127  else
128  gammaY = (k2(pTemp.y(), direction.y()) - pTemp.y()) / direction.y();
129  if (fabs(directionN.z())<epsilon)
130  gammaZ = numeric_limits<float>::max();
131  else
132  gammaZ = (k2(pTemp.z(), direction.z()) - pTemp.z()) / direction.z();
133  gamma = min(min(gammaX, gammaY), gammaZ);
134  pTemp += (gamma) * direction;
135  value += (grid[(int)floor(pTemp.x())+dim[0]*(int)floor(pTemp.y())+(dim[0]*dim[1]*(int)floor(pTemp.z()))]);
136  }
137  value += (grid[(int)floor(p2.x())+dim[0]*(int)floor(p2.y())+(dim[0]*dim[1]*(int)floor(p2.z()))]);
138  return (value / (float)(totalVoxelsCrossed));
139 }
